355  static void fsGenericCallback(uv_fs_t *req) {

358  else { ptr->publish(FsEvent<e>{req->path}); }
359  }

361  template<details::UVFsType e>
362  static void fsResultCallback(uv_fs_t *req) {
363  auto ptr = Request<T, uv_fs_t>::reserve(req);
364  if(req->result < 0) { ptr->publish(ErrorEvent{req->result}); }
365  else { ptr->publish(FsEvent<e>{req->path, static_cast<std::size_t>(req->result)}); }
366  }
367 
368  template<details::UVFsType e>
369  static void fsStatCallback(uv_fs_t *req) {
370  auto ptr = Request<T, uv_fs_t>::reserve(req);
371  if(req->result < 0) { ptr->publish(ErrorEvent{req->result}); }
372  else { ptr->publish(FsEvent<e>{req->path, req->statbuf}); }
373  }
374 
375  static void fsStatfsCallback(uv_fs_t *req) {
376  auto ptr = Request<T, uv_fs_t>::reserve(req);
377  if(req->result < 0) { ptr->publish(ErrorEvent{req->result}); }
378  else { ptr->publish(FsEvent<Type::STATFS>{req->path, *static_cast<Statfs *>(req->ptr)}); }
379  }
380 
381  template<typename... Args>
382  void cleanupAndInvoke(Args&&... args) {
383  uv_fs_req_cleanup(this->get());
384  this->invoke(std::forward<Args>(args)...);
385  }
386 
387  template<typename F, typename... Args>
388  void cleanupAndInvokeSync(F &&f, Args&&... args) {
389  uv_fs_req_cleanup(this->get());
390  std::forward<F>(f)(std::forward<Args>(args)..., nullptr);
391  }

394  using Time = std::chrono::duration<double>;
395  using Type = details::UVFsType;
396  using EntryType = details::UVDirentTypeT;

399 };
