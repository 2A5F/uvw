350  auto ptr = Request<T, uv_fs_t>::reserve(req);
351  if(req->result < 0) { ptr->publish(ErrorEvent{req->result}); }
352  else { ptr->publish(FsEvent<e>{req->path, req->statbuf}); }
353  }

355  template<typename... Args>
356  void cleanupAndInvoke(Args&&... args) {
357  uv_fs_req_cleanup(this->get());
358  this->invoke(std::forward<Args>(args)...);
359  }
360 
361  template<typename F, typename... Args>
362  void cleanupAndInvokeSync(F &&f, Args&&... args) {
363  uv_fs_req_cleanup(this->get());
364  std::forward<F>(f)(std::forward<Args>(args)..., nullptr);
365  }
